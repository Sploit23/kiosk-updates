                this.elements.previewImage?.addEventListener('touchstart', (e) => {
                    if (e.touches.length === 2) {

                            e.touches[0].pageY - e.touches[1].pageY
                        );

                            e.touches[0].pageY - e.touches[1].pageY
                        );
                        
                        if (touchDistance > touchStartDistance * 1.1) {
                            e.target.classList.add('zoomed');
                        } else if (touchDistance < touchStartDistance * 0.9) {
                            e.target.classList.remove('zoomed');

            handleKeyboardNavigation(key) {

                    this.selectPhoto(photoIds[currentIndex + 1]);
                    this.sounds.select();
                } else if (key === 'ArrowLeft' && currentIndex > 0) {
                    this.selectPhoto(photoIds[currentIndex - 1]);
                    this.sounds.select();

            processImages(imageData) {
                if (typeof imageData === 'object' && !Array.isArray(imageData)) {
                    this.groups = imageData;
                } else {
                    this.groups = {};
                    imageData.forEach(name => {
                        const parts = name.split('_');
                        const num = parts[parts.length - 1].replace(/\..+$/, '');
                        
                        if (!this.groups[num]) {
                            this.groups[num] = [];
                        }
                        
                        this.groups[num].push(name);
                    });
                }
            }

                        this.selectPhoto(this.currentPhoto);
                    } else {
                        this.selectPhoto(photoIds[0]);

                }
